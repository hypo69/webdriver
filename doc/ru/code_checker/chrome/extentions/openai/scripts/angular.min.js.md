            $scope.activeTab = tab;
        };

        $scope.isTabActive = function (tab) {
            return $scope.activeTab === tab;
        };

        $scope.sendMessage = async function () {
            console.debug("Отправка сообщения:", $scope.message);
            try {

            } catch (error) {

            }
          
        };

        $scope.trainModel = function () {
